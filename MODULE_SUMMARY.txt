USAGE EXAMPLES:
────────────────────────────────────────────────────────────────────────────────

Basic (3 required variables):
  module "tags" {
    source = "./modules/cloudtags"
    environment = "dev"
    application = "api"
    owner       = "api-team"
  }

Complete (all variables):
  module "tags" {
    source      = "./modules/cloudtags"
    environment = "prod"
    application = "webapp"
    owner       = "platform"
    cost_center = "CC-001"
    team        = "ops"
    business_unit = "product"
    
    additional_tags = {
      Backup = "daily"
    }
    
    mandatory_tags = ["Environment", "Owner", "CostCenter"]
  }

AWS EC2:
  resource "aws_instance" "web" {
    tags = module.tags.tags
    # Name = "${module.tags.name_prefix}-web-001"
  }

AWS ASG:
  resource "aws_autoscaling_group" "app" {
    tag = module.tags.tags_list
  }

Azure:
  resource "azurerm_resource_group" "rg" {
    tags = module.tags.azure_tags
  }

GCP:
  resource "google_compute_instance" "vm" {
    labels = module.tags.gcp_labels
  }
